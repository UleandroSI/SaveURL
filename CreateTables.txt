CREATE TABLE `saveurl`.`cadastro` ( `ID` INT NOT NULL AUTO_INCREMENT COMMENT 'Chave primaria e chave estrangeira em urls.' , `name` VARCHAR(40) NOT NULL COMMENT 'Utilizado para criar o link do endereço.' , `link` VARCHAR(32) NOT NULL COMMENT 'Endereço único da página, criado por Hash MD5.' , `created` DATE NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'Data de criação da página.' , PRIMARY KEY (`ID`)) ENGINE = InnoDB; 

CREATE TABLE `saveurl`.`dados` ( `ID` INT COMMENT 'ID herdado da tabela cadastro.' , `urls` VARCHAR(32) NOT NULL COMMENT 'URLs armazenadas.', FOREIGN KEY (ID) REFERENCES cadastro(ID)) ENGINE = InnoDB;
